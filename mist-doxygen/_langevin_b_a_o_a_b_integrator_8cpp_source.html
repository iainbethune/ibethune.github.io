145 void LangevinBAOABIntegrator::Step(double dt)
146 {
147  // Langevin BAOAB Scheme (from Charlie Matthews' NAMD-lite implementation)
148  // for details see http://dx.doi.org/10.1063/1.4802990
149  // constraints implemented by Ralf Banisch
150 
151  double gdt = friction / system->Picosecond() * dt;
152  double c1 = exp(-gdt);
153  double c3 = sqrt(kbt * (1 - c1 * c1));
154 
155  double sqrtinvm;
156  Vector3 v;
157 
158  // Velocity half-step (B)
159  VelocityStep(0.5 * dt);
160  ResolveConstraints_vel(0.5 * dt);
161 
162  // Position half step (A)
163  PositionStep(0.5 * dt);
164  ResolveConstraints_pos(0.5 * dt);
165  ResolveConstraints_vel(0.5 * dt);
166 
167  // Velocity update (O)
168  int t = 0;
169 #pragma omp parallel default(none) private(v, t, sqrtinvm) shared(c1, c3)
170  {
171 #ifdef _OPENMP
172  t = omp_get_thread_num();
173 #endif
174 #pragma omp for
175  for (int i = 0; i < system->GetNumParticles(); i++)
176  {
177  v = system->GetVelocity(i);
178  sqrtinvm = system->GetInverseSqrtMass(i);
179  v.x = c1 * v.x + sqrtinvm * c3 * r[t]->random_gaussian();
180  v.y = c1 * v.y + sqrtinvm * c3 * r[t]->random_gaussian();
181  v.z = c1 * v.z + sqrtinvm * c3 * r[t]->random_gaussian();
182  system->SetVelocity(i, v);
183  }
184  }
185  ResolveConstraints_vel(dt);
186 
187  // Position half step (A)
188  PositionStep(0.5 * dt);
189  ResolveConstraints_pos(0.5 * dt);
190  ResolveConstraints_vel(0.5 * dt);
191 
192  system->UpdateForces();
193 
194  // Velocity half-step (B)
195  VelocityStep(0.5 * dt);
196  ResolveConstraints_vel(0.5 * dt);
197 
198  // write into diagnostic file
199  // constraintSolver->WriteCounts(5, 100);
200 }
