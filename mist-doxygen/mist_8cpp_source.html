90 int MIST_Init()
91 {
92  int err;
93  char *type;
94 
95  if (params == NULL)
96  {
97  IO::Info("Reading parameters from file: %s\n", MIST_PARAM_FILE);
98  // Read and allocate params.
99  err = readParams(MIST_PARAM_FILE, &params);
100 
101  if (err == ERR_NOFILE)
102  {
103  IO::Error("No input file mist.params could be found.\n");
104  return MIST_NO_SETUP_FILE;
105  }
106  else if (err != ERR_OK)
107  {
108  IO::Error("Unknown file open error, type: %d\n", err);
109  return MIST_NO_SETUP_FILE;
110  }
111  }
112  else
113  {
114  IO::Info("Using parameters set by host application\n");
115  }
116 
117  // Check which integrator type has been selected and initialise it.
118  err = getParamStringValue(params, INTEGRATOR_TYPE_LABEL, &type);
119  if (err != ERR_OK)
120  {
121  IO::Error("No integrator type was identified.\n");
122  return MIST_INVALID_INTEGRATOR;
123  }
124  else if (strcasecmp(type, "verlet") == 0)
125  {
126  integrator = new VerletIntegrator(params);
127  }
128  else if (strcasecmp(type, "leapfrog") == 0)
129  {
130  integrator = new LeapfrogIntegrator(params);
131  }
132  else if (strcasecmp(type, "langevin") == 0)
133  {
134  integrator = new LangevinIntegrator(params);
135  }
136  else if (strcasecmp(type, "yoshida4") == 0)
137  {
138  integrator = new Yoshida4Integrator(params);
139  }
140  else if (strcasecmp(type, "yoshida8") == 0)
141  {
142  integrator = new Yoshida8Integrator(params);
143  }
144  else if (strcasecmp(type, "tempering") == 0)
145  {
146  integrator = new ContinuousTempering(params);
147  }
148  else if (strcasecmp(type, "tamd") == 0)
149  {
150  integrator = new TAMD(params);
151  }
152  else if (strcasecmp(type, "tempering_tamd") == 0)
153  {
154  integrator = new ContinuousTempering_TAMD(params);
155  }
156  else
157  {
158  IO::Error("Your choice ( %s ) for the integrator is unknown.\n", type);
159  return MIST_INVALID_INTEGRATOR;
160  }
161 
162  if (integrator == NULL)
163  {
164  IO::Error("Integrator object not allocated.\n");
165  return MIST_INVALID_INTEGRATOR;
166  }
167 
168  // Initialise the system object, attach it to the integrator.
169  System *s = NULL;
170 #if defined(__MIST_WITH_NAMD_LITE)
171  s = new NAMDLiteSystem();
172 #elif defined(__MIST_WITH_FORCITE)
173  s = new ForciteSystem();
174 #elif defined(__MIST_WITH_GROMACS)
175  s = new GromacsSystem();
176 #elif defined(__MIST_WITH_AMBER)
177  if ((integrator->GetFeatures()) & MIST_FEATURE_FORCE_COMPONENTS)
178  {
179  IO::Error("The integrator you have selected requires force "
180  "components, but this option is not available in AMBER.\n");
181  return MIST_INVALID_OPTION;
182  }
183  s = new AmberSystem();
184 #endif
185 
186  if (s == NULL)
187  {
188  IO::Error("System object not allocated.\n");
189  return MIST_INVALID_SYSTEM;
190  }
191 
192  integrator->SetSystem(s);
193 
194  IO::Info("Initialised MIST Library [Version " PACKAGE_VERSION
195  " (" GITVERSION ")]\n");
196  return MIST_OK;
197 }

539 int MIST_GetNDoF(int *ndof)
540 {
541  int n = integrator->GetSystem()->GetTotalNumParticles();
542  int c = integrator->GetNumConstraints();
543 
544  if (integrator->ConservesMomentum())
545  {
546  if (n == 0 || n == 1)
547  {
548  *ndof = 0;
549  }
550  else if (n == 2)
551  {
552  *ndof = 1 - c;
553  }
554  else
555  {
556  if (integrator->GetSystem()->HasPBC())
557  {
558  *ndof = (3 * n - c - 3);
559  }
560  else
561  {
562  *ndof = (3 * n - c - 6);
563  }
564  }
565  }
566  else
567  {
568  *ndof = (3 * n - c);
569  }
570 
571  if (*ndof <= 0)
572  {
573  IO::Error("No degrees of freedom in the system!");
574  return MIST_INVALID_SYSTEM;
575  }
576 
577  return MIST_OK;
