93 double GromacsSystem::GetPotentialEnergy(force_component_t t) const
94 // Set variables needed by GROMACS to compute force component
95 {
96  double e;
97  switch (t)
98  {
99  case (MIST_FORCE_ALL):
100  e = (real)((gmx_enerdata_t *)potential_energy)->term[F_EPOT];
101  break;
102  case (MIST_FORCE_BOND):
103  e = (real)((gmx_enerdata_t *)potential_energy)->term[F_BONDS] +
104  ((gmx_enerdata_t *)potential_energy)->term[F_G96BONDS] +
105  ((gmx_enerdata_t *)potential_energy)->term[F_MORSE] +
106  ((gmx_enerdata_t *)potential_energy)->term[F_CUBICBONDS] +
107  ((gmx_enerdata_t *)potential_energy)->term[F_CONNBONDS] +
108  ((gmx_enerdata_t *)potential_energy)->term[F_HARMONIC] +
109  ((gmx_enerdata_t *)potential_energy)->term[F_FENEBONDS] +
110  ((gmx_enerdata_t *)potential_energy)->term[F_TABBONDS] +
111  ((gmx_enerdata_t *)potential_energy)->term[F_TABBONDSNC] +
112  ((gmx_enerdata_t *)potential_energy)->term[F_RESTRBONDS];
113  break;
114  case (MIST_FORCE_BEND):
115  e = (real)((gmx_enerdata_t *)potential_energy)->term[F_ANGLES] +
116  ((gmx_enerdata_t *)potential_energy)->term[F_G96ANGLES] +
117  ((gmx_enerdata_t *)potential_energy)->term[F_RESTRANGLES] +
118  ((gmx_enerdata_t *)potential_energy)->term[F_LINEAR_ANGLES] +
119  ((gmx_enerdata_t *)potential_energy)->term[F_CROSS_BOND_BONDS] +
120  ((gmx_enerdata_t *)potential_energy)->term[F_CROSS_BOND_ANGLES] +
121  ((gmx_enerdata_t *)potential_energy)->term[F_UREY_BRADLEY] +
122  ((gmx_enerdata_t *)potential_energy)->term[F_QUARTIC_ANGLES] +
123  ((gmx_enerdata_t *)potential_energy)->term[F_TABANGLES];
124  break;
125  case (MIST_FORCE_PROPDIHED):
126  e = (real)((gmx_enerdata_t *)potential_energy)->term[F_PDIHS] +
127  ((gmx_enerdata_t *)potential_energy)->term[F_RBDIHS] +
128  ((gmx_enerdata_t *)potential_energy)->term[F_RESTRDIHS] +
129  ((gmx_enerdata_t *)potential_energy)->term[F_CBTDIHS] +
130  ((gmx_enerdata_t *)potential_energy)->term[F_FOURDIHS] +
131  ((gmx_enerdata_t *)potential_energy)->term[F_CMAP] +
132  ((gmx_enerdata_t *)potential_energy)->term[F_TABDIHS];
133  break;
134  case (MIST_FORCE_IMPROPDIHED):
135  e = (real)((gmx_enerdata_t *)potential_energy)->term[F_IDIHS] +
136  ((gmx_enerdata_t *)potential_energy)->term[F_PIDIHS];
137  break;
138  case (MIST_FORCE_NONBONDED):
139  e = GetPotentialEnergy(MIST_FORCE_ALL) -
140  GetPotentialEnergy(MIST_FORCE_PROPDIHED) -
141  GetPotentialEnergy(MIST_FORCE_IMPROPDIHED) -
142  GetPotentialEnergy(MIST_FORCE_BEND) -
143  GetPotentialEnergy(MIST_FORCE_BOND);
144  break;
145  default:
146  e = 0.0;
147  break;
148  }
149 
150  return e;
151 }

158 Vector3 GromacsSystem::GetForce(int i, force_component_t t) const
159 {
160  real x, y, z;
161 
162  switch (t)
163  {
164  case (MIST_FORCE_ALL):
165  x = ((force_arrays_t *)forces)->f_total[i][0];
166  y = ((force_arrays_t *)forces)->f_total[i][1];
167  z = ((force_arrays_t *)forces)->f_total[i][2];
168  break;
169  case (MIST_FORCE_BOND):
170  x = ((force_arrays_t *)forces)->f_bond[i][0];
171  y = ((force_arrays_t *)forces)->f_bond[i][1];
172  z = ((force_arrays_t *)forces)->f_bond[i][2];
173  break;
174  case (MIST_FORCE_BEND):
175  x = ((force_arrays_t *)forces)->f_angle[i][0];
176  y = ((force_arrays_t *)forces)->f_angle[i][1];
177  z = ((force_arrays_t *)forces)->f_angle[i][2];
178  break;
179  case (MIST_FORCE_PROPDIHED):
180  x = ((force_arrays_t *)forces)->f_propdihed[i][0];
181  y = ((force_arrays_t *)forces)->f_propdihed[i][1];
182  z = ((force_arrays_t *)forces)->f_propdihed[i][2];
183  break;
184  case (MIST_FORCE_IMPROPDIHED):
185  x = ((force_arrays_t *)forces)->f_impropdihed[i][0];
186  y = ((force_arrays_t *)forces)->f_impropdihed[i][1];
187  z = ((force_arrays_t *)forces)->f_impropdihed[i][2];
188  break;
189  case (MIST_FORCE_NONBONDED):
190  x = ((force_arrays_t *)forces)->f_nonbonded[i][0];
191  y = ((force_arrays_t *)forces)->f_nonbonded[i][1];
192  z = ((force_arrays_t *)forces)->f_nonbonded[i][2];
193  break;
194  default:
195  x = 0.0;
196  y = 0.0;
197  z = 0.0;
198  break;
199  }

204 void GromacsSystem::SetMassPtr(void *m, int massrequirements)
205 {
206  int i;
207 
208  if (numParticles < 1) return;
209 
210  if (massrequirements & MIST_MASS_REQD)
211  {
212  if (masses != NULL)
213  {
214  delete[] masses;
215  }
216  masses = new double[numParticles];
217  for (i = 0; i < numParticles; i++)
218  {
219  masses[i] = 1. / (((real *)m)[i]);
220  }
221  }
222 
223  if (massrequirements & MIST_INV_MASS_REQD)
224  {
225  if (invmasses != NULL)
226  {
227  delete[] invmasses;
228  }
229  invmasses = new double[numParticles];
230  for (i = 0; i < numParticles; i++)
231  {
232  invmasses[i] = (((real *)m)[i]);
233  }
234  }
235 
236  if (massrequirements & MIST_INVSQRT_MASS_REQD)
237  {
238  if (invsqrtmasses != NULL)
239  {
240  delete[] invsqrtmasses;
241  }
242  invsqrtmasses = new double[numParticles];
243  for (i = 0; i < numParticles; i++)
244  {
245  invsqrtmasses[i] = sqrt(double(((real *)m)[i]));
246  }
247  }
248 }
