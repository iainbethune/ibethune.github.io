126 /* return a number from a standard gaussian distribution */
127 double MIST_Random::random_gaussian()
128 {
129  double fac, rsq, v1, v2;
130 
131  if (second_gaussian_waiting)
132  {
133  second_gaussian_waiting = 0;
134  return second_gaussian;
135  }
136  else
137  {
138  /*
139  * rsq >= 1.523e-8 ensures abs result < 6
140  * make sure we are within unit circle
141  */
142  do
143  {
144  v1 = 2.0 * random_uniform() - 1.0;
145  v2 = 2.0 * random_uniform() - 1.0;
146  rsq = v1 * v1 + v2 * v2;
147  } while (rsq >= 1. || rsq < 1.523e-8);
148  fac = sqrt(-2.0 * log(rsq) / rsq);
149  /*
150  * Now make the Box-Muller transformation to get two normally
151  * distributed random numbers. Save one and return the other.
152  */
153  second_gaussian_waiting = 1;
154  second_gaussian = v1 * fac;
155  return v2 * fac;
156  }
157 }
