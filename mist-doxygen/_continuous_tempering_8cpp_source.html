359 void ContinuousTempering::Step(double dt)
360 {
361  // Langevin BAOAB Scheme (from CM's NAMD-lite implementation)
362  double gdt = friction / system->Picosecond() *
363  dt; // dimensionless friction constant (friction is in ps^{-1})
364  double c1 = exp(-gdt);
365  double c3 = sqrt(kbt * (1 - c1 * c1));
366  // additional deg. of freed. (=ss) and other necessary var.
367  static int iter = 0;
368  static double ss = 0; // 0;
369  static double v_ss = 0;
370  static double force_ss = 0; // Also this must be dec. as static since it is
371  // needed for the first half (B) kick!
372  static double coupl = 0,
373  d_coupl = 0; // static for the same reason as force_ss
374  double Delta = Delta_inp;
375  double Delta2 = Delta2_inp;
376  double DeltaWall = DeltaWall_inp;
377  double Tfact = Tfact_inp;
378  // convert from kj/mol input energy value to MDengine energy units
379  double pe_offset =
380  pe_offset_inp * system->Boltzmann() / BOLTZMANN_KJoverMOL;
381  static double m_ss = m_ss_inp * system->AMU();
382  double fric_ss = fric_ss_inp;
383 
384  // BAOAB xi variables and service variables
385  double g_ssdt = fric_ss / system->Picosecond() *
386  dt; // dimensionless friction constant for xi
387  double c1_ss = exp(-g_ssdt);
388  double c3_ss = sqrt(kbt * (1 - c1_ss * c1_ss));
389  double l_char_inv =
390  1 / system->Angstrom(); // this is 1/[1 angstrom] in internal units
391  int i;
392  double m_inv;
393  Vector3 v, p, f, ftemp, r3;
394 
395  // Velocity half-step (B)
396  for (i = 0; i < system->GetNumParticles(); i++)
397  {
398  m_inv = system->GetInverseMass(i);
399  v = system->GetVelocity(i);
400  f = system->GetForce(i);
401  f = Vector3::Scale(1 - coupl, f); // rescale forces with coupling
402  v = v + Vector3::Scale(dt * 0.5 * m_inv, f); // update velocity
403  system->SetVelocity(i, v);
404  }
405  ResolveVelocityConstraints(0.5 * dt);
406 
407  // velocity step for additional dof. For unit consistency, need to divide
408  // force_ss by a length scale [in internal units]
409  v_ss += dt * 0.5 / m_ss * force_ss * l_char_inv;
410 
411  // Position half step (A)
412  PositionStep(0.5 * dt);
413  ResolvePositionConstraints(0.5 * dt);
414  ResolveVelocityConstraints(0.5 * dt);
415  ss += l_char_inv * dt * 0.5 * v_ss;
416 
417  // Velocity update (O)
418  for (i = 0; i < system->GetNumParticles(); i++)
419  {
420  m_inv = system->GetInverseMass(i);
421  v = system->GetVelocity(i);
422  r3 = Vector3(r->random_gaussian(), r->random_gaussian(),
423  r->random_gaussian());
424  v = Vector3::Scale(c1, v) + Vector3::Scale(sqrt(m_inv) * c3, r3);
425  system->SetVelocity(i, v);
426  }
427  ResolveVelocityConstraints(dt);
428  // Velocity update for xi
429  v_ss = c1_ss * v_ss + sqrt(1.0 / m_ss) * c3_ss * r->random_gaussian();
430 #ifdef __MIST_WITH_MPI
431  // In parallel, we just take the master's value i.e. v_ss is equal on all
432  // processes
433  MPI_Bcast(&v_ss, 1, MPI_DOUBLE, 0, system->GetCommunicator());
434 #endif
435 
436  // Position half step (A)
437  PositionStep(0.5 * dt);
438  ResolvePositionConstraints(0.5 * dt);
439  ResolveVelocityConstraints(0.5 * dt);
440  ss += l_char_inv * dt * 0.5 * v_ss;
441 
442  system->UpdateForces();
443 
444  // compute the force force_ss acting on xi
445  SS_coupling(&coupl, &d_coupl, ss, Delta, Delta2, Tfact);
446  // must be initialized to zero as functions only add contributions
447  force_ss = 0;
448  static_SSforce(&force_ss, ss, coupl, d_coupl, Delta, Delta2, pe_offset);
449  // compute metadynamics contribution to f_ss
450  if (iter % meta_dep == 0)
451  metadyn(&force_ss, ss, 1, Delta, Delta2);
452  else
453  metadyn(&force_ss, ss, 0, Delta, Delta2);
454 
455  // Velocity half-step (B)
456  for (i = 0; i < system->GetNumParticles(); i++)
457  {
458  m_inv = system->GetInverseMass(i);
459  v = system->GetVelocity(i);
460  f = system->GetForce(i);
461  f = Vector3::Scale(1 - coupl, f); // rescale forces with coupling
462  v = v + Vector3::Scale(dt * 0.5 * m_inv, f); // update velocity
463  system->SetVelocity(i, v);
464  }
465  ResolveVelocityConstraints(0.5 * dt);
466  // velocity step for additional dof. For unit consistency, need to divide
467  // force_ss by a length scale [in internal units]
468  v_ss += dt * 0.5 / m_ss * force_ss * l_char_inv;
469 
470  // I think this is necessary to be synchronized with Gromacs when saving
471  // conf.
472  if (iter % save_conf == save_conf - 1)
473  write_confxyz(coupl, ss, Delta, iter + 1, force_ss * l_char_inv, f.x);
474  iter += 1;
475 }

477 void ContinuousTempering::SS_coupling(double *cc, double *d_cc, double s,
478  double D, double D2, double Tf)
479 {
480  // This function evaluates the value f(xi) of the coupling function and
481  // returns it as 'cc', and
482  // it evaluates the derivative d/dxi f(xi) and returns it as 'd_cc'. f(xi)
483  // is a piecewise polynomial function which
484  // is zero in the interval [-D,D], takes the value 'Tf' if |s|>D2 and
485  // smoothly interpolates in between. See
486  // the Continuous Tempering paper for more information.
487  if (std::abs(s) <= D)
488  {
489  *cc = 0;
490  *d_cc = 0;
491  }
492  else if ((D < std::abs(s)) && (std::abs(s) < D + D2))
493  {
494  *cc = Tf * (3 * pow((std::abs(s) - D) / D2, 2) -
495  2 * pow((std::abs(s) - D) / D2, 3));
496  if (s > 0)
497  *d_cc = 6 * Tf * (((std::abs(s) - D) / D2) -
498  pow((std::abs(s) - D) / D2, 2)) /
499  D2;
500  else
501  *d_cc = -6 * Tf * (((std::abs(s) - D) / D2) -
502  pow((std::abs(s) - D) / D2, 2)) /
503  D2;
504  }
505  else
506  {
507  *cc = Tf;
508  *d_cc = 0;
509  }
510 }

512 void ContinuousTempering::static_SSforce(double *fs, double s, double cc,
513  double d_cc, double D, double Dl,
514  double peoff)
515 {
516  // this function evaluates the RHS of the eom for xi, i.e.
517  // fs = d/dxi f(xi)*V(q,p) + d/dxi Phi(xi),
518  // with f(xi) = coupling function and Phi(xi) = confining potential.
519  //
520  // contrib. due to coupling
521  double pe = system->GetPotentialEnergy(); // read potential energy
522  *fs += d_cc * (pe - peoff); // V_extended_sys=(V-peoff)-cc*(V-peoff)
523  // term compensating approximately for temperature change
524  *fs += -d_cc * cc / (1 - cc) * 18738.0; // kbt*2500.0*3.2;
525 }

527 void ContinuousTempering::metadyn(double *fs, double s, bool addgauss, double D,
528  double D2)
529 {
530 // This function deals with the metadynamics part of Continuous Tempering.
531 // It does three things:
532 // (1) The metadynamics biasing force (i.e. the derivative of the biasing
533 // potential 'VG') is evaluated and
534 // returned as 'fs'. (2) If addgauss==1, then the metadynamics potential
535 // 'VG' is modified by adding another
536 // gaussian with height 'hills_height' and width 'sigma' at the current
537 // position 's' of the xi variable.
538 // (3) If this function is called for the first time and read_bias_file==1,
539 // then the potential 'VG' is read
540 // from the file 'meta.0'.
541 //
542 #ifdef __MIST_WITH_MPI
543  // Do everything on the rank 0 in MPI, then broadcast the result
544  int rank;
545  MPI_Comm_rank(system->GetCommunicator(), &rank);
546  if (rank == 0)
547  {
548 #endif
549  // gaussian height and width:
550  static double hills_height = hills_height_inp * system->Boltzmann() /
551  BOLTZMANN_KJoverMOL,
552  sigma = hills_width_inp;
553  // how many gaussians are deposited before saving bias potential to
554  // file:
555  const int metastride = 1000;
556  const int NG = 9000;
557  static double VG[NG + 2] = {0};
558  static int it = 0;
559  const double smax = 1.03 * (D + D2);
560  const double smin = -smax;
561  const double dsg = (smax - smin) / (NG - 1);
562  const double swp = smax + dsg;
563  const double swm = smin - dsg;
564  static bool isfirst = 1;
565 
566  // read input bias potential
567  if (isfirst && (read_bias_file == 1))
568  {
569  isfirst = 0;
570  std::ifstream biasfile;
571  biasfile.open("meta.0");
572  int count_lines = 0;
573  std::string inputline;
574  while (getline(biasfile, inputline))
575  count_lines++;
576  if ((NG + 2) != count_lines)
577  {
578  IO::Error("Incompatible number of lines in input bias "
579  "potential file meta.0 "
580  "number of lines must equal the size of the internal "
581  "grid which is %d\n",
582  NG + 2);
583  exit(0);
584  }
585  biasfile.clear();
586  biasfile.seekg(0);
587  for (int ig = 0; ig < NG + 2; ig++)
588  {
589  double cc;
590  biasfile >> cc >> VG[ig];
591  }
592  biasfile.close();
593  }
594 
595  // add gaussian
596  if (addgauss)
597  {
598  it = it + 1;
599  for (int ig = 1; ig < NG + 1; ig++)
600  {
601  VG[ig] = VG[ig] +
602  hills_height *
603  exp(-0.5 *
604  pow((s - (swm + (ig - 1) * dsg)) / sigma, 2)) +
605  // adding the same hill on the symmetric position
606  hills_height *
607  exp(-0.5 *
608  pow((-s - (swm + (ig - 1) * dsg)) / sigma, 2));
609  }
610  // write metadyn bias potential
611  if (it % metastride == 0)
612  {
613  char filename[15];
614  sprintf(filename, "meta.%d", (it / metastride));
615  std::ofstream biasfile;
616  biasfile.open(filename);
617  for (int ig = 0; ig < NG + 2; ig++)
618  {
619  biasfile << std::setw(15) << swm + (ig - 1) * dsg
620  << std::setw(20) << VG[ig] << std::endl;
621  }
622  biasfile.close();
623  }
624  }
625 
626  // evaluate bias force
627  if ((s > swm) && (s < swp))
628  {
629  // Bin evaluation
630  int ig = (int)ceil((s - swm) / dsg);
631  // evaluate force
632  if ((ig >= 1) && (ig < NG))
633  *fs += -(VG[ig + 1] - VG[ig]) / dsg;
634  }
635  // confining potential term at the border
636  if (std::abs(s) > 1.03 * (D + D2))
637  {
638  *fs += -(s / std::abs(s)) * 25.0 * kbt *
639  (std::abs(s) - 1.03 * (D + D2)) / D;
640  }
641 
642 #ifdef __MIST_WITH_MPI
643  }
644  // Broadcast the biasing force computed on rank zero
645  MPI_Bcast(fs, 1, MPI_DOUBLE, 0, system->GetCommunicator());
646 #endif
647 }

649 void ContinuousTempering::write_confxyz(double cc, double s, double D, int it,
650  double fss, double fx)
651 {
652 // this function writes the position of xi and the value of the coupling
653 // function
654 // for the current frame into the file 'coupling.dat'. This happens every
655 // save_conf
656 // frames.
657 
658 #ifdef __MIST_WITH_MPI
