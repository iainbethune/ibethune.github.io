50 {
51  // Yoshida 8th order algorithm
52 
53  int j, k;
54  double d[8] = {0.104242620869991e1, 0.182020630970714e1,
55  0.157739928123617e0, 0.244002732616735e1,
56  -0.716989419708120e-2, -0.244699182370524e1,
57  -0.161582374150097e1, -0.17808286265894516e1};
58 
59  Vector3 v, p, f;
60 
61  for (k = 0; k < 15; k++)
62  {
63  if (k < 8)
64  j = k;
65  else
66  j = 14 - k;
67 
68  // Velocity half-step
69  VelocityStep(0.5 * dt * d[j]);
70 
71  // Position full step
72  PositionStep(dt * d[j]);
73
