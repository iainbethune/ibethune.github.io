178 void NAMDLiteSystem::SetMassPtr(void *m, int massrequirements)
179 {
180  int i;
181 
182  if (numParticles < 1) return;
183 
184  if (massrequirements & MIST_MASS_REQD)
185  {
186  masses = (double *)m;
187  }
188 
189  if (massrequirements & MIST_INV_MASS_REQD)
190  {
191  invmasses = new double[numParticles];
192  for (i = 0; i < numParticles; i++)
193  {
194  invmasses[i] = 1. / (((double *)m)[i]);
195  }
196  }
197 
198  if (massrequirements & MIST_INVSQRT_MASS_REQD)
199  {
200  invsqrtmasses = new double[numParticles];
201  for (i = 0; i < numParticles; i++)
202  {
203  invsqrtmasses[i] = 1. / sqrt(((double *)m)[i]);
204  }
205  }
