177 double System::ComputeDihedrals(int phi_ind[])
178 {
179  // compute the dihedral angle defined by the atoms i,j,k,l
180  // see Arnaud Blondel and Martin Karplus, 'New Formulation for Derivatives
181  // of Torsion Angles and Improper Torsion Angles in Molecular Mechanics:
182  // Elimination of Singularities', JCC 1995,
183  // http://onlinelibrary.wiley.com/doi/10.1002/.
184  int i = phi_ind[0];
185  int j = phi_ind[1];
186  int k = phi_ind[2];
187  int l = phi_ind[3];
188  double qb_norm;
189  Vector3 qa, qb, qc, n1, n2;
190  qa = GetPosition(i) - GetPosition(j);
191  qb = GetPosition(j) - GetPosition(k);
192  qc = GetPosition(l) - GetPosition(k);
193  n1 = Vector3::Cross(qa, qb);
194  n2 = Vector3::Cross(qc, qb);
195  qb_norm = sqrt(Vector3::Mult(qb, qb));
196  return atan2(Vector3::Mult(Vector3::Cross(n2, n1), qb) / qb_norm,
197  Vector3::Mult(n1, n2));
198 }

203 Vector3 System::ComputeDihedralsJac(int phi_ind[], int n)
204 {
205  // compute the Jacobian (first derivative) d\phi/dq_n of dihedral angle \phi
206  // defined by the atoms i,j,k,l with respect to q_n.
207  // see Arnaud Blondel and Martin Karplus, 'New Formulation for Derivatives
208  // of Torsion Angles and Improper Torsion Angles in Molecular Mechanics:
209  // Elimination of Singularities', JCC 1995,
210  // http://onlinelibrary.wiley.com/doi/10.1002/.
211  int i = phi_ind[0];
212  int j = phi_ind[1];
213  int k = phi_ind[2];
214  int l = phi_ind[3];
215  double qb_norm;
216  Vector3 qa, qb, qc, n1, n2;
217  if ((n == i) || (n == j) || (n == k) || (n == l))
218  {
219  qa = GetPosition(i) - GetPosition(j);
220  qb = GetPosition(j) - GetPosition(k);
221  qc = GetPosition(l) - GetPosition(k);
222  n1 = Vector3::Cross(qa, qb);
223  n2 = Vector3::Cross(qc, qb);
224  n1 = Vector3::Scale(1.0 / Vector3::Mult(n1, n1), n1);
225  n2 = Vector3::Scale(1.0 / Vector3::Mult(n2, n2), n2);
226  qb_norm = sqrt(Vector3::Mult(qb, qb));
227  if (n == i)
228  {
229  return Vector3::Scale(-1 * qb_norm, n1);
230  }
231  else if (n == j)
232  {
233  return Vector3::Scale(qb_norm + (Vector3::Mult(qa, qb) / qb_norm),
234  n1) -
235  Vector3::Scale(Vector3::Mult(qb, qc) / qb_norm, n2);
236  }
237  else if (n == k)
238  {
239  return Vector3::Scale((Vector3::Mult(qb, qc) / qb_norm) - qb_norm,
240  n2) -
241  Vector3::Scale(Vector3::Mult(qa, qb) / qb_norm, n1);
242  }
243  else if (n == l)
244  {
245  return Vector3::Scale(qb_norm, n2);
246  }
247  }
248 
249  // nothing to do, derivative is zero.
250  return Vector3(0, 0, 0);
251 }

253 Vector3 System::WrapVectorIntoCell(const Vector3 a)
254 {
255  // this function wraps a Vector3 object back inside the cell. It is
256  // appropriate e.g. for position vectors.
257  if (!pbc)
258  return a;
259 
260  Vector3 image;
261  image.x = a.x - cell_a.x * floor(a.x * cell_inv.x);
262  image.y = a.y - cell_b.y * floor(a.y * cell_inv.y);
263  image.z = a.z - cell_c.z * floor(a.z * cell_inv.z);
264  return image;
265 }

267 Vector3 System::MinimumImageDiff(Vector3 ai, const Vector3 aj)
268 {
269  // this function applies pbc's to a distance vector da = a(i) - a(j) in such
270  // a way that a(i) is fixed and a(j) is chosen according to the minimum
271  // image
272  // convention. Distances can't be greater then half the cell dimension in
273  // each
274  // direction.
275  ai = ai - aj;
276  if (!pbc)
277  return ai;
278 
279  Vector3 image;
280  image.x = ai.x - cell_a.x * round(ai.x * cell_inv.x);
281  image.y = ai.y - cell_b.y * round(ai.y * cell_inv.y);
282  image.z = ai.z - cell_c.z * round(ai.z * cell_inv.z);
283  return image;
284 }
