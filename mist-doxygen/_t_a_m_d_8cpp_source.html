236 void TAMD::SetSystem(System *s)
237 {
238  Integrator::SetSystem(s);
239  kbt = temp * system->Boltzmann();
240  // scale masses and spring constant to the right units
241  kappa = kappa_inp * system->Boltzmann() / BOLTZMANN_KJoverMOL;
242  m_s = m_s_inp * system->AMU() * pow(system->Angstrom(), 2);
243  sqrt_ms_inv = sqrt(1.0 / m_s);
244  // the mass of the additional dofs needs to have unit [mL^2] to balance the
245  // fact that the s variables are dimensionless.
246  kbt_s = temp_s * system->Boltzmann();
247 
248  if (constraintSolver->IsConstraintsOn())
249  {

257 void TAMD::TAMDvars_PositionStep(double dt)
258 {
259  // position step for additional dofs.
260  s_phi += dt * v_phi;
261  s_psi += dt * v_psi;
262 
263  // take into account periodicity of angular variables. If outside of
264  // [-pi,pi],
265  // we push s_phi and s_psi back in
266  if (s_phi > M_PI)
267  s_phi -= TWO_TIMES_PI;
268  if (s_phi < (-1 * M_PI))
269  s_phi += TWO_TIMES_PI;
270  if (s_psi > M_PI)
271  s_psi -= TWO_TIMES_PI;
272  if (s_psi < (-1 * M_PI))
273  s_psi += TWO_TIMES_PI;

278 double TAMD::DihedralDiff(double phi1, double phi2)
279 {
280  if (std::abs(phi1 - phi2) <= M_PI)
281  {
282  // phi1 and phi2 are both in [-pi,pi], nothing needs to be done
283  return kappa * (phi1 - phi2);
284  }
285  else if (phi1 < phi2)
286  {
287  // translate phi1 by 2*pi.
288  return kappa * (phi1 + TWO_TIMES_PI - phi2);
289  }
290  else
291  {
292  // translate phi1 by -2*pi.
293  return kappa * (phi1 - TWO_TIMES_PI - phi2);
294  }

301 void TAMD::Step(double dt)
302 {
303  // TAMD algorithm
304  // for details see http://dx.doi.org/10.1063/1.4802990
305  double m, sqrtinvm;
306  static int iter = 0;
307  int i;
308  Vector3 v;
309  if (iter == 0)
310  {
311  // set BAOAB constants
312  gdt = friction / system->Picosecond() * dt;
313  c1 = exp(-gdt);
314  c3 = sqrt(kbt * (1 - c1 * c1));
315  // set BAOAB constants for additional dofs
316  g_sdt = fric_s / system->Picosecond() * dt;
317  c1_s = exp(-g_sdt);
318  c3_s = sqrt(kbt_s * (1 - c1_s * c1_s));
319  // initialise additional dof positions and velocities
320  // additional dof associated to phi
321  s_phi = system->ComputeDihedrals(phi_ind);
322  v_phi = 0;
323  force_phi = 0;
324  // additional dof associated to psi
325  s_psi = system->ComputeDihedrals(psi_ind);
326  v_psi = 0;
327  force_psi = 0;
328  }
329  //*** end of variable declarations ***
330 
331  //*** beginning of iteration ***
332  // Velocity half-step (B)
333  // velocity step for system vars
334  VelocityStep(0.5 * dt);
335 
336  // contribution to velocities due to coupling to additional dofs
337  for (i = 0; i < 4; i++)
338  {
339  // the phi coupling contribution to system velocities
340  m = system->GetInverseMass(phi_ind[i]);
341  v = system->GetVelocity(phi_ind[i]);
342  v = v -
343  Vector3::Scale(0.5 * dt * m * force_phi,
344  system->ComputeDihedralsJac(phi_ind, phi_ind[i]));
345  system->SetVelocity(phi_ind[i], v);
346  // the psi coupling contribution to system velocities
347  m = system->GetInverseMass(psi_ind[i]);
348  v = system->GetVelocity(psi_ind[i]);
349  v = v -
350  Vector3::Scale(0.5 * dt * m * force_psi,
351  system->ComputeDihedralsJac(psi_ind, psi_ind[i]));
352  system->SetVelocity(psi_ind[i], v);
353  }
354 
355  // velocity step for additional dofs.
356  v_phi += dt * 0.5 / m_s * force_phi;
357  v_psi += dt * 0.5 / m_s * force_psi;
358 
359  // Position half step (A)
360  PositionStep(0.5 * dt);
361 
362  // position step for additional dofs.
363  TAMDvars_PositionStep(0.5 * dt);
364 
365  // Velocity update (O)
366  for (int i = 0; i < system->GetNumParticles(); i++)
367  {
368  m = system->GetInverseMass(i);
369  v = system->GetVelocity(i);
370  sqrtinvm = sqrt(m);
371  v.x = c1 * v.x + sqrtinvm * c3 * r->random_gaussian();
372  v.y = c1 * v.y + sqrtinvm * c3 * r->random_gaussian();
373  v.z = c1 * v.z + sqrtinvm * c3 * r->random_gaussian();
374  system->SetVelocity(i, v);
375  }
376 
377  // Velocity update (O) for additional dofs
378  v_phi = c1_s * v_phi + sqrt_ms_inv * c3_s * r->random_gaussian();
379  v_psi = c1_s * v_psi + sqrt_ms_inv * c3_s * r->random_gaussian();
380 
381  // Position half step (A)
382  PositionStep(0.5 * dt);
383 
384  // position step for additional dofs.
385  TAMDvars_PositionStep(0.5 * dt);
386 
387  // force updates
388 
389  system->UpdateForces();
390 
391  // compute forces due to coupling, taking periodicy into account
392  force_phi = DihedralDiff(system->ComputeDihedrals(phi_ind), s_phi);
393  force_psi = DihedralDiff(system->ComputeDihedrals(psi_ind), s_psi);
394 
395  // Velocity half-step (B)
396  VelocityStep(0.5 * dt);
397 
398  // contribution to velocities due to coupling to additional dofs
399  for (i = 0; i < 4; i++)
400  {
401  // the phi coupling contribution to system velocities
402  m = system->GetInverseMass(phi_ind[i]);
403  v = system->GetVelocity(phi_ind[i]);
404  v = v -
405  Vector3::Scale(0.5 * dt * m * force_phi,
406  system->ComputeDihedralsJac(phi_ind, phi_ind[i]));
407  system->SetVelocity(phi_ind[i], v);
408  // the psi coupling contribution to system velocities
409  m = system->GetInverseMass(psi_ind[i]);
410  v = system->GetVelocity(psi_ind[i]);
411  v = v -
412  Vector3::Scale(0.5 * dt * m * force_psi,
413  system->ComputeDihedralsJac(psi_ind, psi_ind[i]));
414  system->SetVelocity(psi_ind[i], v);
415  }
416 
417  // velocity step for additional dofs.
418  v_phi += dt * 0.5 / m_s * force_phi;
419  v_psi += dt * 0.5 / m_s * force_psi;
420 
421  // *** end of time step *** //
422 
423  // write force out
424  static std::ofstream outfile;
425  // File initialization
426  if (iter == 0)
427  outfile.open("tamd_out.dat");
428  // write force to file dihedrals.dat
429  if (iter % save_freq == 0)
430  {
431  outfile << force_phi << "\t" << force_psi << "\t" << s_phi << "\t"
432  << s_psi << std::endl;
433  outfile.flush();
434  }
435  // if (iter % 100 == 0) write_confxyz(coupl, xi);
436  iter += 1;
437 }
